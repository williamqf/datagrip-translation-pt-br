Relata problemas em contextos baseados em XML do Spring Webflow:

  • Referências de bean não resolvidas
  • Tags ou atributos obrigatórios ausentes
  • Tipos de propriedade incorretos
  • Propriedades enum inconsistentes
  • Tipo incorreto de bean referenciado

Exemplo:


  <beans ...
    xmlns:webflow="http://www.springframework.org/schema/webflow-config ...">
    
    <webflow:flow-executor id="flowExecutorWithoutFlowRegistryRef"/> <!-- Não foi possível encontrar <flow-registry> com o ID padrão 'flowRegistry' -->
    <webflow:flow-executor
      id=""  <!-- O valor não deve estar vazio -->
      flow-registry="INVALID_VALUE"  <!-- Não foi possível resolver o bean INVALID_VALUE -->
           <webflow:flow-execution-repository
                   conversation-manager="dummyBean" <!-- O bean deve ser do tipo 'org.springframework.webflow.conversation.ConversationManager' -->
                    max-executions="INVALID_VALUE" <!-- O valor deve ser um número inteiro -->
                    max-execution-snapshots="INVALID_VALUE"/>  <!-- O valor deve ser um número inteiro -->
           <webflow:flow-execution-listeners>
			<webflow:listener criteria="*"
                           ref="dummyBean" /> <!-- O bean deve ser do tipo 'org.springframework.webflow.execution.FlowExecutionListener' -->
           </webflow:flow-execution-listeners>
    </webflow:flow-executor>
    <webflow:flow-executor id="dummy"
      flow-registry="dummyBean"/> <!-- O bean deve ser do tipo 'org.springframework.webflow.definition.registry.FlowDefinitionRegistry' -->

    <bean id="dummyBean" class="java.lang.String"/>
  </beans>
