Relata métodos de repositório CRUD do Spring Data para os quais o construtor de consultas Spring Data Query não consegue gerar uma consulta correspondente.

Exemplo:


interface PersonRepository extends Repository<Person, Long> {
  List<Person> findByEmailAddressAndLastname(EmailAddress emailAddress, String lastname);
  List<Person> findByEmailUnknownAndLastname(EmailAddress emailAddress, String lastname); // Não foi possível resolver a propriedade 'Unknown'
  List<Person> findByAndLastname(EmailAddress emailAddress, String lastname); // Nome da propriedade ausente

  List<Person> findDistinctPeopleByLastnameOrFirstname(String lastname, String firstname);
  List<Person> findPeopleDistinctByUnknownOrFirstname(String lastname, String firstname); // Não foi possível resolver a propriedade 'Unknown'

  List<Person> findByLastnameOrderByFirstnameAsc(String lastname);
  List<Person> findByLastnameOrderByUnknownDesc(String lastname);  // Não foi possível resolver a propriedade 'Unknown'
}
