Relata o uso de this em closures para referenciar propriedades de um contexto externo.

Exemplo:


function Outer() {
  this.outerProp = 1;
  function inner() {
    // Inadequado, pois o 'outerProp' de Outer 
    // não será atualizado aqui como esperado
    // ao chamar 'new Outer()'
    this.outerProp = 2;
  }
  inner();
}
