Relata referências de Bean incorretas no parâmetro scheduler da anotação @Scheduled.

Exemplo:


  @Component public class UserController {...}
  @Component public class UserServiceAsyncTaskScheduler extends SimpleAsyncTaskScheduler {...}

  @Component public class UserService {
    @Scheduled(scheduler="userServiceAsyncTaskScheduler", cron="1 * * * * * ?")
    public void scheduleTask() {...}

    @Scheduled(scheduler="userController", cron="1 * * * * * ?") // Relata "O Bean deve ser de um dos seguintes tipos: TaskScheduler, ScheduledExecutorService"
    public void scheduleTask_2() {...}

    @Scheduled(scheduler="unknown", cron="1 * * * * * ?")   // Relata "Não é possível resolver o bean 'unknown'"
    public void scheduleTask_3() {...}
  }
