Relata instruções import redundantes.

Importações padrão e não utilizadas podem ser removidas com segurança.

Exemplo:


  import kotlin.*
  import kotlin.collections.*
  import kotlin.comparisons.*
  import kotlin.io.*
  import kotlin.ranges.*
  import kotlin.sequences.*
  import kotlin.text.*

  // específico para jvm
  import java.lang.*
  import kotlin.jvm.*

  // específico para js
  import kotlin.js.*

  import java.io.* // esta importação não é utilizada e pode ser removida
  import java.util.*

  fun foo(list: ArrayList<String>) {
      list.add("")
  }
