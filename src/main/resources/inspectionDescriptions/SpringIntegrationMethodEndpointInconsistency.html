Relata declarações de métodos de endpoint do Spring Integration incorretas.

Exemplo:


class MyEndpoints {
  @InboundChannelAdapter("channel")
  public void cannotBeVoid() {...} // O método anotado com @InboundChannelAdapter deve ter um tipo de retorno

  @InboundChannelAdapter("channel")
  public String cannotHaveParams(String s) {..} // O método anotado com @InboundChannelAdapter não pode ter argumentos

  @Filter(inputChannel = "channel", // O endpoint pode ter apenas um poller
    outputChannel = "channel2",
    poller = {@Poller(fixedDelay = "100"), @Poller(fixedRate = "100")})
  public void testMultiplePollers() {
  }

  @Filter(inputChannel = "channel",
  outputChannel = "channel2",
  poller = @Poller(value = "poller", maxMessagesPerPoll = "100"))
  public void testValue() {
  }

  @Filter(inputChannel = "channel",
    outputChannel = "channel2",
    poller = @Poller(trigger = "trigger", cron = "0 */10 * * * *")) // O atributo 'trigger' é mutuamente exclusivo com outros atributos
  public void testTrigger() {
  }
}
