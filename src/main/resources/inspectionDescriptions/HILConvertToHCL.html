Relata expressões HCL de estilo antigo e sugere a conversão para expressões HCL2.

Exemplo:


  # Configuração do Terraform 0.11. Algumas partes foram omitidas para clareza.
resource "azurerm_app_service" "example" {
 count               = "${length(local.app_services)}"
 name                = "${lower(lookup(local.app_services[count.index], "kind"))}-appservice"
 location            = "${azurerm_resource_group.example.location}"
 resource_group_name = "${azurerm_resource_group.example.name}"
 app_service_plan_id = "${element(azurerm_app_service_plan.example.*.id, count.index)}"

 site_config {
   # omitido para clareza
 }
}

Após aplicar a correção rápida:


  # Configuração do Terraform 0.12. Algumas partes foram omitidas para clareza.
resource "azurerm_app_service" "example" {
 count               = length(local.app_services)
 name                = "${lower(local.app_services[count.index].kind)}-appservice"
 location            = azurerm_resource_group.example.location
 resource_group_name = azurerm_resource_group.example.name
 app_service_plan_id = azurerm_app_service_plan.example[count.index].id

 site_config {
    # omitido para clareza
 }
}
