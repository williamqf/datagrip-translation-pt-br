Relata aspectos e pointcuts incorretos. Quando determinados advices de AOP correspondem a beans de aspecto, isso pode causar erros em tempo de execução.

Exemplo: 


<beans>
  <aop:config>
    <aop:pointcut expression="args()"/> <!-- O pointcut deve ter um ID -->
    <aop:aspect/> <!-- O atributo 'ref' deve ser definido -->
  </aop:config>
</beans>

<beans>
  <bean id="aspect1" class="Aspect1"/>
  <bean id="aspect2" class="Aspect2"/>
  <aop:config>
    <aop:aspect ref="aspect1"> <!-- Alguns advices correspondem a este bean -->
    <aop:before method="xxx" pointcut="execution(* foo())"/>
    </aop:aspect>
    <aop:aspect ref="aspect2">
      <aop:before method="xxx" pointcut="execution(* foo())"/>
    </aop:aspect>
  </aop:config>
</beans>
