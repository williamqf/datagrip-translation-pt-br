Relata métodos @EventListener incorretos.

Exemplo:


  @Configuration @ComponentScan
  open class Config

  data class MyEvent(val string: String)

  @Component
  class LogComponent {
    @EventListener // Métodos anotados com @EventListener devem ser públicos
    private fun logCommands(commandName: MyEvent) {}

    @EventListener // O método deve ter no máximo um parâmetro
    fun processCommand(commandName: MyEvent, text: String) {}
  }
