Relata nomes de Bean duplicados definidos no contexto da aplicação.

Exemplo:


  <beans profile="PROD">
    <bean class="..." id="duplicatedName"/>  <!-- Erro -->
    <bean class="..." name="foo_name, duplicatedName"/> <!-- Erro em 'duplicatedName' -->
    <bean class="..." id="toBeAliased"/>

    <alias alias="duplicatedName" name="toBeAliased"/> <!-- Erro em 'duplicatedName' -->
  </beans>
