Relata beans de esquema incorretos definidos no contexto de aplicação XML.


  <beans ... xmlns:util="https://www.springframework.org/schema/util">
    <bean class="MyBean" id="myBean"/>
    <bean class="ErrTypeBean" id="errTypeBean"/>
    <util:list list-class="java.lang.String"/> // Relata "A classe deve ser uma subclasse de 'java.util.List'"
    <util:set value-type="MyBean">
      <ref bean="fooBean"/>
      <ref bean="errTypeBean"/> // Relata "Não é possível adicionar a uma coleção do tipo 'MyBean'"
    </util:set>
  </beans>
