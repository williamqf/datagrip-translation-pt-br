Relata servlets que não possuem um mapeamento associado.

Exemplo:


  <web-app>
    <servlet>
      <servlet-name>foo</servlet-name> // Erro: o servlet deve ter um mapeamento
      <servlet-class>javax.servlet.Servlet</servlet-class>
    </servlet>
  
    <filter>
      <filter-name>bar</filter-name> // Erro: o filtro deve ter um mapeamento
      <filter-class>java.lang.String</filter-class>
    </filter>
  </web-app>
