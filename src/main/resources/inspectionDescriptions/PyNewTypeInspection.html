Relata usos inválidos de NewType.

Exemplo: 


  from typing import NewType

  InvalidName = NewType("Name", int)  # o nome da variável 'InvalidName' não corresponde ao nome do NewType 'Name'


  from typing import Literal

  InvalidType = NewType("InvalidType", Literal[1])  # NewType não pode ser usado com 'Literal[1]'


  Base = NewType("Base", str)

  class Derived(Base):  # 'Base' não pode ser subclassificada
      pass
