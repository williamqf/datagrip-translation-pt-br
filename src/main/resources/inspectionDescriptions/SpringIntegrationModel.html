Relata problemas em contextos de aplicação do Spring Integration baseados em XML:

  • Referências de Bean não resolvidas
  • Tags ou atributos obrigatórios ausentes
  • Tipos de propriedade incorretos
  • Propriedades enum inconsistentes
  • Tipo incorreto do Bean referenciado

Exemplo:


  <beans ... >
  <int:channel id="channel"> <!-- 'dispatcher' e 'queue' não podem ser usados juntos  -->
    <int:dispatcher failover="true"/>
    <int:priority-queue capacity="4"/>
  </int:channel>
  
  <int:channel id="another" >
    <int:queue <!-- Exatamente um entre 'message-store' ou 'ref' é obrigatório  -->
              capacity="4" message-store="myMessageStore" ref="myQueue"/>
  </int:channel>
  
   <int:gateway id="asyncGatewayError"
                async-executor="dummyBean"/> <!-- O Bean deve ser do tipo 'java.util.concurrent.Executor'  -->

   <int:delayer default-delay="3"
              message-store="myMessageStore"
              scheduler="scheduler"
              wait-for-tasks-to-complete-on-shutdown="false"/>
  <!-- O Bean deve ser do tipo 'java.util.concurrent.Executor'  -->
  <int:delayer
    default-delay="INVALID_VALUE"  <!-- Não é possível converter a string 'INVALID_VALUE' para a classe de destino 'java.lang.Integer' -->
               message-store="dummyBean" <!-- O Bean deve ser do tipo 'org.springframework.integration.store.MessageStore' -->
               scheduler="dummyBean" <!-- O Bean deve ser um destes tipos: java.util.concurrent.ScheduledExecutorService, org.springframework.scheduling.TaskScheduler -->
               wait-for-tasks-to-complete-on-shutdown="INVALID_VALUE"/> <!-- Não foi possível resolver o símbolo 'INVALID_VALUE' -->

  <bean id="dummyBean" class="java.lang.String"/>
  <bean id="myMessageStore" class="org.springframework.integration.store.SimpleMessageStore"/>
  <task:scheduler id="scheduler"/>
</beans>
