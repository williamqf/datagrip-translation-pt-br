Relata elementos de advice e advisor incorretos se os atributos pointcut ou pointcut-ref estiverem ausentes.

Exemplo:


<beans>
    <aop:config>
      <aop:pointcut id="zzz" expression="args()"/>
        <aop:aspect ref="xxx">
            <aop:before method="writeArg1" pointcut="execution(* *(String))"/>
            <aop:after method="aaa" pointcut-ref="zzz"/>
            <aop:after-throwing method="aaa"/> <!-- O atributo 'pointcut' ou 'pointcut-ref' deve ser definido -->
       </aop:aspect>
    </aop:config>
</beans>
