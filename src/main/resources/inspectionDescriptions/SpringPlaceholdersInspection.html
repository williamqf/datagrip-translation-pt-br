Relata placeholders não resolvidos configurados no contexto XML do Spring.

Exemplo:


  <beans>
    <ctx:property-placeholder location="application.properties"/>

    <bean class="beans.ServerComponent">
      <property name="serverName" value="${server.name}}"/>
      <property name="port" value="${unknown.property}}"/> <!-- Relata "Não é possível resolver a chave da propriedade" -->
    </bean>
  </beans>
