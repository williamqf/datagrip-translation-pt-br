Relata conflitos de configuração nos atributos de <bean>.

Exemplo:


  <beans>
    <bean class="java.lang.String" factory-bean="factoryBean"/>
    <!-- Apenas um entre 'class' ou 'factory-bean' é necessário -->
    <bean factory-bean="factoryBean" id="missingFactoryMethod"/> <!-- Quando factory-bean estiver presente, factory-method é necessário -->

    <!-- O atributo 'abstract' com o valor padrão 'false' é supérfluo -->
    <bean abstract="false" id="superfluousDefaultAttributeAbstract"/>

    <!-- O atributo 'scope' com o valor padrão 'singleton' é supérfluo -->
    <bean id="superfluousDefaultAttributeScope" scope="singleton"/>

    <bean class="java.lang.StringBuilder">
      <property name="length">
        <!-- Os atributos 'id', 'name' e 'scope' são supérfluos -->
        <bean id="beanId" name="beanName" scope="prototype"/>
      </property>
    </bean>
  </beans>
