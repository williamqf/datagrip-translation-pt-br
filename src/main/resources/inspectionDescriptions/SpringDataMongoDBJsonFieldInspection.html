Relata campos de documento não resolvidos especificados na anotação org.springframework.data.mongodb.repository.Query.

Exemplo:


@Document
class User {
  @Field
  String name;
}

interface UserRepository extends Repository<User, Long> {
  @Query("{ age: { $gt: ?0 } }") // 'age' é destacado como não resolvido
  List<User> findUsersOlderThan(Integer age);
}
