Relata chamadas de função with redundantes que não acessam nada do receptor.

Exemplo:


  class MyClass {
      fun f(): String = ""
  }

  fun testRedundant() {
      with(c) { // <== 'with' é redundante porque 'c' não é usado
          println("1")
      }
  }

  fun testOk() {
      val c = MyClass()
      with(c) { // <== OK porque 'f()' é efetivamente 'c.f()'
          println(f())
      }
  }
