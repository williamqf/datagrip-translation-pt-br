Relata o uso de VirtualFile.getChildren() em métodos recursivos.

Isso pode causar um loop infinito ao iterar sobre links simbólicos cíclicos. Em vez disso, use VfsUtilCore.visitChildrenRecursively(). 


void processDirectory(VirtualFile dir) {
  for (VirtualFile file : dir.getChildren()) { // Incorreto
    if (!file.isDirectory()) {
      processFile(file);
    } else {
      processDirectory(file); // Chamada recursiva
    }
  }
}



void processDirectory(VirtualFile dir) {
  VfsUtilCore.visitChildrenRecursively(dir, new VirtualFileVisitor<Void>() { // Correto
    @Override
    public boolean visitFile(@NotNull VirtualFile file) {
      if (!file.isDirectory()) {
        processFile(file);
      }
      return true;
    }
  });
}
